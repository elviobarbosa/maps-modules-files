import { Injectable } from "@angular/core";
import { MapsActionUsecase } from "maps-angular-components";
import { Observable } from "rxjs";

import { ParametrizacaoAplicacaoResgateRepository } from "../repository/parametrizacao-aplicacao-resgate.repository";

@Injectable({
  providedIn: "root",
})
export class ParametrizacaoAplicacaoResgateSaveUsecase extends MapsActionUsecase<
  any,
  any
> {
  constructor(private readonly _repository: ParametrizacaoAplicacaoResgateRepository) {
    super();
  }

  protected action = (obj: any): Observable<any> => {
    return this._repository.saveNatfin(obj);
  };
}
