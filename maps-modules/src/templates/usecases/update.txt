import { Injectable } from '@angular/core';
import { MapsActionUsecase } from 'maps-angular-components';

import { Observable } from 'rxjs';
import { ParametrizacaoAplicacaoResgateRepository } from '../repository/parametrizacao-aplicacao-resgate.repository';
import { ParametrizacaoAplicacaoResgateEditBodyOutput, ParametrizacaoAplicacaoResgateResponse } from '../entities/parametrizacao-aplicacao-resgate-resquest-response.entity';


@Injectable({
  providedIn: 'root'
})
export class ParametrizacaoAplicacaoResgateEditUsecase extends MapsActionUsecase<
ParametrizacaoAplicacaoResgateEditBodyOutput, ParametrizacaoAplicacaoResgateResponse> {
    
  constructor(private readonly _repository: ParametrizacaoAplicacaoResgateRepository) {
    super();
  }

  protected action = (data: ParametrizacaoAplicacaoResgateEditBodyOutput): Observable<ParametrizacaoAplicacaoResgateResponse> => {
    return this._repository.editNatfinParametrizacao(data);
  };
}
